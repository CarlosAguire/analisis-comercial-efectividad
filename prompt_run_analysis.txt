import json
import ast

# 1. Asegurar conversión a string puro
raw_str = JSON_TABLE.decode('utf-8', errors='ignore') if isinstance(JSON_TABLE, bytes) else str(JSON_TABLE)
raw_str = raw_str.strip().strip('"""').strip("'''").strip()

data_list = []
try:
    data_list = json.loads(raw_str)
except Exception:
    try:
        # Limpieza de caracteres de escape
        data_list = json.loads(ast.literal_eval(f'"{raw_str}"'))
    except Exception as e:
        print(f"Error crítico de parseo: {e}")
        data_list = []

# 2. Limpieza estructural estricta
for row in data_list:
    row.pop("@odata.etag", None)
    row.pop("ItemInternalId", None)
    
    # Normalizar notas nulas/vacías a string vacío
